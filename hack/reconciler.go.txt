const (
	errNot<Resource> = "managed resource is not a <Resource>"
	errNewClient     = "cannot build Anthropic client"
	errObserve       = "cannot observe <Resource>"
	errCreate        = "cannot create <Resource>"
	errUpdate        = "cannot update <Resource>"
	errDelete        = "cannot delete/archive <Resource>"
)

// Setup adds a controller for <Resource> to the supplied manager. When
// skipDefaultMetadata is false (the default), it registers an initializer
// that writes canonical Crossplane identifiers into
// spec.forProvider.metadata. Pass-through only — the initializer requires
// that <Resource>Parameters has a `Metadata map[string]string` field.
// If this resource has no such field, drop the skipDefaultMetadata
// parameter (and the conditional below) from both Setup and SetupGated.
func Setup(mgr ctrl.Manager, o controller.Options, skipDefaultMetadata bool) error {
	name := managed.ControllerName(betav1alpha1.<Resource>Kind)

	opts := []managed.ReconcilerOption{
		managed.WithExternalConnector(&connector{kube: mgr.GetClient()}),
		managed.WithLogger(o.Logger.WithValues("controller", name)),
		managed.WithPollInterval(o.PollInterval),
		managed.WithManagementPolicies(),
	}
	if !skipDefaultMetadata {
		opts = append(opts, managed.WithInitializers(initializer.New(mgr.GetClient(), "metadata")))
	}

	return ctrl.NewControllerManagedBy(mgr).
		Named(name).
		WithOptions(o.ForControllerRuntime()).
		For(&betav1alpha1.<Resource>{}).
		Complete(managed.NewReconciler(mgr,
			resource.ManagedKind(betav1alpha1.<Resource>GroupVersionKind),
			opts...,
		))
}

// SetupGated registers the <Resource> controller to start only once the
// <Resource> CRD is established.
func SetupGated(mgr ctrl.Manager, o controller.Options, skipDefaultMetadata bool) error {
	o.Gate.Register(func() {
		if err := Setup(mgr, o, skipDefaultMetadata); err != nil {
			panic(err)
		}
	}, betav1alpha1.<Resource>GroupVersionKind)
	return nil
}

// connector builds an ExternalClient for each reconcile.
type connector struct {
	kube client.Client
}

func (c *connector) Connect(ctx context.Context, mg resource.Managed) (managed.ExternalClient, error) {
	res, ok := mg.(*betav1alpha1.<Resource>)
	if !ok {
		return nil, xperrors.New(errNot<Resource>)
	}

	// clients.NewClient resolves the ProviderConfig (namespaced or cluster-scoped),
	// tracks usage, extracts credentials, and constructs the Anthropic client.
	cl, err := clients.NewClient(ctx, c.kube, res)
	if err != nil {
		return nil, xperrors.Wrap(err, errNewClient)
	}

	return &external{client: cl}, nil
}

// external implements managed.ExternalClient for Anthropic <Resource>s.
type external struct {
	client *anthropic.Client
}

func (e *external) Observe(ctx context.Context, mg resource.Managed) (managed.ExternalObservation, error) {
	res, ok := mg.(*betav1alpha1.<Resource>)
	if !ok {
		return managed.ExternalObservation{}, xperrors.New(errNot<Resource>)
	}

	resID := meta.GetExternalName(res)
	if resID == res.GetName() {
		return managed.ExternalObservation{ResourceExists: false}, nil
	}

	resp, err := e.client.Beta.<Resource>s.Get(ctx, resID, anthropic.Beta<Resource>GetParams{})
	if err != nil {
		var apiErr *anthropic.Error
		if errors.As(err, &apiErr) && apiErr.StatusCode == 404 {
			return managed.ExternalObservation{ResourceExists: false}, nil
		}
		return managed.ExternalObservation{}, xperrors.Wrap(err, errObserve)
	}

	// TODO: check ArchivedAt if applicable
	// if resp.ArchivedAt != "" { return managed.ExternalObservation{ResourceExists: false}, nil }

	// Populate AtProvider.
	res.Status.AtProvider.ID = &resp.ID
	// TODO: populate remaining AtProvider fields

	res.SetConditions(xpv1.Available())

	return managed.ExternalObservation{
		ResourceExists:   true,
		ResourceUpToDate: isUpToDate(res, resp),
	}, nil
}

func (e *external) Create(ctx context.Context, mg resource.Managed) (managed.ExternalCreation, error) {
	res, ok := mg.(*betav1alpha1.<Resource>)
	if !ok {
		return managed.ExternalCreation{}, xperrors.New(errNot<Resource>)
	}

	params := buildNewParams(res.Spec.ForProvider)
	resp, err := e.client.Beta.<Resource>s.New(ctx, params)
	if err != nil {
		return managed.ExternalCreation{}, xperrors.Wrap(err, errCreate)
	}

	meta.SetExternalName(res, resp.ID)
	res.Status.AtProvider.ID = &resp.ID

	return managed.ExternalCreation{}, nil
}

func (e *external) Update(ctx context.Context, mg resource.Managed) (managed.ExternalUpdate, error) {
	res, ok := mg.(*betav1alpha1.<Resource>)
	if !ok {
		return managed.ExternalUpdate{}, xperrors.New(errNot<Resource>)
	}

	resID := meta.GetExternalName(res)
	if resID == res.GetName() {
		return managed.ExternalUpdate{}, xperrors.New("external name not yet set; skipping update")
	}

	params := buildUpdateParams(res.Spec.ForProvider)
	if _, err := e.client.Beta.<Resource>s.Update(ctx, resID, params); err != nil {
		return managed.ExternalUpdate{}, xperrors.Wrap(err, errUpdate)
	}

	return managed.ExternalUpdate{}, nil
}
